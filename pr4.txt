 #include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define SIZE 3

typedef struct Node {
    int mat[SIZE][SIZE];
    int x, y;              // blank position
    int level;
    struct Node* parent;
} Node;

int row[] = {1, 0, -1, 0};
int col[] = {0, -1, 0, 1};

int isSafe(int x, int y) {
    return (x >= 0 && x < SIZE && y >= 0 && y < SIZE);
}

Node* newNode(int mat[SIZE][SIZE], int x, int y, int newX, int newY, int level, Node* parent) {
    Node* node = (Node*)malloc(sizeof(Node));
    memcpy(node->mat, mat, sizeof(node->mat));

    // swap blank
    int temp = node->mat[x][y];
    node->mat[x][y] = node->mat[newX][newY];
    node->mat[newX][newY] = temp;

    node->x = newX;
    node->y = newY;
    node->level = level;
    node->parent = parent;

    return node;
}

int isGoal(int mat[SIZE][SIZE], int goal[SIZE][SIZE]) {
    return memcmp(mat, goal, sizeof(int) * SIZE * SIZE) == 0;
}

void printPath(Node* node) {
    if (node == NULL) return;
    printPath(node->parent);

    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++)
            printf("%d ", node->mat[i][j]);
        printf("\n");
    }
    printf("\n");
}

void solve(int initial[SIZE][SIZE], int x, int y, int goal[SIZE][SIZE]) {
    Node* queue[10000];
    int front = 0, rear = 0;

    Node* root = newNode(initial, x, y, x, y, 0, NULL);
    queue[rear++] = root;

    while (front < rear) {
        Node* curr = queue[front++];

        if (isGoal(curr->mat, goal)) {
            printf("Solution Found:\n\n");
            printPath(curr);
            return;
        }

        for (int i = 0; i < 4; i++) {
            int newX = curr->x + row[i];
            int newY = curr->y + col[i];

            if (isSafe(newX, newY)) {
                Node* child = newNode(curr->mat, curr->x, curr->y, newX, newY, curr->level + 1, curr);
                queue[rear++] = child;
            }
        }
    }

    printf("No Solution\n");
}

int main() {
    int initial[SIZE][SIZE] = {
        {1, 2, 3},
        {5, 6, 0},
        {7, 8, 4}
    };

    int goal[SIZE][SIZE] = {
        {1, 2, 3},
        {5, 8, 6},
        {7, 0, 4}
    };

    int x = 1, y = 2; // blank position (0)

    solve(initial, x, y, goal);

    return 0;
}
